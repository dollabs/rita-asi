
function getTeamMembers(team){
	firebase.database().ref("teams/"+team+"/members").once('value', function(snapshot) {
		return snapshot.val();
	});
}

function getAllMembers(){
	firebase.database().ref("teams/").once('value', function(snapshot) {
		console.log("getting members")
		var allMembers = [];
		var teams = snapshot.val();
		for (i=0; i< Object.keys(teams).length; i++){
			var team = teams[i];
			var members = getTeamMembers(team);
			allMembers.concat(members);
		}
		return allMembers
	});
}
